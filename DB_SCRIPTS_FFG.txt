CREATE TABLE TBL_JOB_INFORMATION
(	JOB_ID INT NOT NULL,
	COMPANY_NAME VARCHAR(50) NOT NULL,

	JOB_DESCRIPTION VARCHAR(1000),
	CITY_ID INT NOT NULL,
	VACANCY_COUNT INT NOT NULL,
	QUALIFICATION_REQ VARCHAR(100),
	CREATE_TIMESTAMP TIMESTAMP NOT NULL,
	CONSTRAINT TBL_JOB_INFORMATION_PK PRIMARY KEY( JOB_ID),
